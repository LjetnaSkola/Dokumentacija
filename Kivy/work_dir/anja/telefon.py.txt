import kivy
from kivy.app import App
from kivy.uix.label import Label
from kivy.uix.button import Button
from kivy.uix.textinput import TextInput
from kivy.uix.gridlayout import GridLayout

class Telefon(App):
    def build(self):

        layout = GridLayout(cols=3, rows=5)
        
        self.inputs = [] 
        input_field = TextInput(multiline=False)
        self.inputs.append(input_field)
        layout.add_widget(input_field)
        layout.add_widget(TextInput())  # Prazna ćelija
        layout.add_widget(TextInput())  # Prazna ćelija

        btn1 = Button(text='1 (,.?!)')
        layout.add_widget(btn1)
        btn2 = Button(text='2 (abc)')
        layout.add_widget(btn2)
        btn3 = Button(text='3 (def)')
        layout.add_widget(btn3)

        btn4 = Button(text='4 (ghi)')
        layout.add_widget(btn4)
        btn5 = Button(text='5 (jkl)')
        layout.add_widget(btn5)
        btn6 = Button(text='6 (mno)')
        layout.add_widget(btn6)

        btn7 = Button(text='7 (pqrs)')
        layout.add_widget(btn7)
        btn8 = Button(text='8 (tuv)')
        layout.add_widget(btn8)
        btn9 = Button(text='9 (wxyz)')
        layout.add_widget(btn9)

        btn_star = Button(text='*')
        layout.add_widget(btn_star)
        btn0 = Button(text='0')
        layout.add_widget(btn0)
        btn_hash = Button(text='#')
        layout.add_widget(btn_hash)

        return layout

Telefon().run()
